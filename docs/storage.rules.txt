rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {

    match /resources/{customerId}/{projectId}/{fileName} {
      allow read, write: if request.auth != null
        && firestore.exists(
          /databases/(default)/documents/admins/$(request.auth.uid)
        );
    }

    match /library/{libraryItemId}/{fileName} {
      allow read, write: if request.auth != null
        && firestore.exists(
          /databases/(default)/documents/admins/$(request.auth.uid)
        );
    }

    match /customerUploads/{customerId}/{uploadId}/{fileName} {
      allow read, write: if request.auth != null
        && firestore.exists(
          /databases/(default)/documents/admins/$(request.auth.uid)
        );

      // A customer may create (not overwrite or delete) a file in their own
      // folder, one file at a time, each 20 MB or smaller. The cumulative
      // 500 MB per-customer total is NOT enforced here — an earlier version
      // tried to check it via a second cross-service Firestore lookup
      // (firestore.get on customers/{customerId}), but that consistently
      // caused every upload to be denied even for customers well under quota,
      // and it could not be diagnosed further without live rules testing.
      // The 500 MB total is instead enforced at the app layer: the Firestore
      // `library` create rule caps each individual record's declared size at
      // 20 MB, and the admin can see each customer's running total on their
      // detail panel in the Console and intervene if someone is clearly over.
      allow write: if request.auth != null
        && resource == null
        && customerId == firestore.get(
             /databases/(default)/documents/customerAccess/$(request.auth.token.email)
           ).data.customerId
        && request.resource.size <= 20 * 1024 * 1024;

      // A customer may read back their own uploaded files (e.g. from their
      // Your Uploads list in the Portal).
      allow read: if request.auth != null
        && customerId == firestore.get(
             /databases/(default)/documents/customerAccess/$(request.auth.token.email)
           ).data.customerId;
    }

    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
